2.2     Get Cart Items from Order

1.   Open shared > services > firebase-cart-service.ts.
Add the method getCartOrderById().

firebase-cart.service.ts

  getCartOrderById(cartId: string) {

    const promise = new Promise<CartItem[]>((resolve, reject) => {

      const db = firebase.firestore();

 

      // Read from the DB

      const cartRef = db.collection('carts/' + cartId + '/items');

      cartRef.get().then(itemsSnapshot => {

      const cart: CartItem[] = []; // Empty cart

      itemsSnapshot.forEach(doc => {

        const p = new Product(doc.data().name, doc.data().price, '');

        p.id = doc.id;

        const c = new CartItem(p, doc.data().quantity);

        cart.push(c); // Add to cart

      });

      resolve(cart); // Return cart as result of promise

      }).catch(error => { });

    });

    return promise;

  }

3.   Get the cart items after scanning the QR code for the order id.

orders.page.ts

import { CartItem } from '../shared/models/cartitem';

...

export class OrdersPage implements OnInit {

  orderItems: CartItem[];

  orderId = '';

 

  ...

 

  scan() {

     this.barcodeScanner.scan().then(barcodeData => {

     this.orderId = barcodeData.text;

      this.retrieveOrder(this.orderId);

     }).catch(err => {

         alert('Error ' + err);

     });

  }

 

  retrieveOrder(orderId) {

    this.cartService.getCartOrderById(orderId).then(items => {

      this.orderItems = items;

    });

  }

 

}

3.   Encode the order cart id as QR code when the user clicks Check Out.

cart.page.ts

  checkout() {

    this.cartService.checkout().then((cartId) => {

 

      // Refresh the cart after check out

      this.cartService.getCartItems().then(

        result =>  this.cart = result

      );

 

      // Show a QR code

      this.barcodeScanner.encode(this.barcodeScanner.Encode.TEXT_TYPE, cartId).then(result => {

         console.log('Encode QR ' + result);

      })

      .catch(err => {

        alert('Error ' + err);

      });

 

    });

  }
