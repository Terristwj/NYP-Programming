3.    In the constructor of the class ListPage, add 3 items to the products array.

list.page.ts

import { Component, OnInit } from '@angular/core';

import { Product } from '../shared/models/product';

 

@Component({

  selector: 'app-list',

  templateUrl: 'list.page.html',

  styleUrls: ['list.page.scss']

})

export class ListPage implements OnInit {

  products: Product[];

 

  constructor() {

    this.products = [

      new Product('Coffee', 5.9, 'assets/image/coffee.jpeg'),

      new Product('Rainbow Shake', 7.2, 'assets/image/mermaid.jpg'),

      new Product('Taco', 4.5, 'assets/image/taco.jpeg'),

    ];

  }

 

  ngOnInit() {

  }

 

}

 

2.1   List Item *ngFor

1.    Modify the <ion-list>to show the products.
Add for loop in HTML to repeat <ion-items> inside <ion-list>.
Each item has a label showing the name.

list.page.html

...

<ion-content>

 

  <ion-list>

    <ion-item *ngFor="let item of products">

      <ion-label>

        <h2> {{item.name}} </h2>

     </ion-label>

    </ion-item>

  </ion-list>

 

</ion-content>

2.2   Format Text using Pipe

1.    Add currency pipe to display numbers with $ and 2 decimal places.

list.page.html

...

<ion-content>

 

  <ion-list>

    <ion-item *ngFor="let item of products">

      <ion-label>

        <h2>{{item.name}}</h2>

        <p> {{item.price | currency}} </p>

      </ion-label>

    </ion-item>

  </ion-list>

 

</ion-content>
