1.5   Add to Cart

1.    Open the file src > app > shared > services > cart.service.ts.
Add the following code in the add() method.

cart.service.ts

…

export class CartService {

  …

 

  add(product: Product) {

    // If product already in cart, increase the quantity

    let found = false;

    for (const index in this.cart) {

      if (this.cart[index].product === product) {

        this.cart[index].quantity++;

        found = true;

        break;

      }

    }

 

    // If product not in cart, add with quantity 1

    if (!found) {

      this.cart.push(new CartItem(product, 1));

    }

  }

 

}

product.service.ts

import { Injectable } from '@angular/core';

import { Product } from '../models/product';

 

@Injectable({

  providedIn: 'root'

})

export class ProductService {

 

  allProducts: Product[] = [];

 

  constructor() {

    this.allProducts = [

      new Product('Coffee', 5.9, 'assets/image/coffee.jpeg'),

      new Product('Rainbow Shake', 7.2, 'assets/image/mermaid.jpg'),

      new Product('Taco', 4.5, 'assets/image/taco.jpeg'),

      new Product('Sandwich', 5.5, 'assets/image/sandwich.jpg'),

      new Product('Burger', 6.7, 'assets/image/burger.jpg'),

    ];

  }

 

  getAllProducts() {

    const promise = new Promise<Product[]>((resolve, reject) => {

      resolve(this.allProducts);

    });

    return promise;

  }

 

}
