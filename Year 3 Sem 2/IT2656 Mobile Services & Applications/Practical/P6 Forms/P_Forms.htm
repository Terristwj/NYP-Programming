3    Submit Form

1.    Download the zip file from Blackboard. They are the image placeholder files. Place into ‘assets/image’ folder.

2.    Open the file src > app > shared > models > product.ts
Add 2 more properties category and vegeterian.

product.ts

export class Product {

    name: string;

    price: number;

    image: string;

    category: string;

    vegeterian: boolean;

 

    constructor(name: string, price: number, image: string,

                category?: string, vegeterian?: boolean) {

        this.name = name;

        this.price = price;

        this.image = image;

        this.category = category;

        this.vegeterian = vegeterian;

 

        // Add an image placeholder if there's no image

        if (this.image === undefined) {

            switch (this.category) {

                case 'Main':

                    this.image = 'assets/image/main.png';

                    break;

                case 'Beverage':

                    this.image = 'assets/image/beverage.png';

                    break;

                case 'Dessert':

                    this.image = 'assets/image/dessert.png';

                    break;

                default:

                    this.image = 'assets/icon/icon.png';

            }

        }

    }

}

4    Form Validation

4.1      Required Validation

1.    Open the file src > app > add-product > add-product.ts
Only add to the product service if the user inputs are valid.

add.product.page.ts

import { FormGroup, FormControl, Validators } from '@angular/forms';

…

export class AddProductPage implements OnInit {

  addProductForm: FormGroup;

  categories: string[];

  submitted: boolean;

 

  constructor(private productService: ProductService) {

    this.submitted = false;

    this.categories = ['Main', 'Beverage', 'Dessert'];

    this.addProductForm = new FormGroup({

      name: new FormControl('', [Validators.required]),

      price: new FormControl(0),

      category: new FormControl('Main'),

      vegeterian: new FormControl(false)

   });

  }

 

  ngOnInit() {}

 

  add() {

    this.submitted = true;

 

    if (this.addProductForm.valid) {

      const prod = new Product(this.addProductForm.value.name,

                               this.addProductForm.value.price,

                               undefined,

                               this.addProductForm.value.category,

                               this.addProductForm.value.vegeterian);

      this.productService.add(prod);

    }

  }

 

}

export class AddProductPage implements OnInit {

  addProductForm: FormGroup;

  submitted: boolean;

  categories: string[];

 

  static positiveNumber(fc: FormControl) {

    if (fc.value <= 0) {

      return ({positiveNumber: true});

    } else {

      return (null);

    }

  }

 

  constructor(private formBuilder: FormBuilder,

              private productService: ProductService) {

 

    this.submitted = false;

    this.categories = ['Main', 'Beverage', 'Dessert'];

    this.addProductForm = new FormGroup({

      name: new FormControl('', [Validators.required]),

      price: new FormControl(0, [AddProductPage.positiveNumber]),

      category: new FormControl('Main'),

      vegeterian: new FormControl(false)

   });

  }

 

  ngOnInit() { }

 

  add() { … }

 

}
